26  package de.saly.elasticsearch.riverstate;
27  
28  import javax.mail.Folder;
29  import javax.mail.Message;
30  import javax.mail.MessagingException;
31  import javax.mail.internet.MimeMessage;
32  
33  import org.codehaus.jackson.map.ObjectMapper;
34  import org.codehaus.jackson.type.TypeReference;
35  import org.elasticsearch.action.get.GetResponse;
36  import org.elasticsearch.client.Client;
37  import org.elasticsearch.common.logging.ESLogger;
38  import org.elasticsearch.common.logging.ESLoggerFactory;
39  
40  public class ElasticsearchRiverStateManager implements RiverStateManager {
41  
42      private static final String ERRORS_ID = "errors";
43      private static final String FOLDERSTATE_ID = "folderstate";
44      private static final String RIVERSTATE_TYPE = "imapriverstate";
45      private Client client;
46      private String index;
47      private final ObjectMapper mapper = new ObjectMapper();
48      protected final ESLogger logger = ESLoggerFactory.getLogger(this.getClass().getName());
49  
50      public ElasticsearchRiverStateManager client(final Client client) {
51          this.client = client;
52          return this;
53      }
54  
55      @Override
56      public synchronized RiverState getRiverState(final Folder folder) throws MessagingException {
57  
58          try {
59  
60              if (client.admin().indices().prepareExists(index()).execute().actionGet().isExists()) {
61  
62                  final GetResponse response = client
63                          .prepareGet(index(), RIVERSTATE_TYPE, FOLDERSTATE_ID + "_" + folder.getURLName().toString().hashCode()).execute()
64                          .get();
65  
66                  if (!response.isSourceEmpty()) {
67                      return mapper.readValue(response.getSourceAsString(), new TypeReference<RiverState>() {
68                      });
69  
70                  }
71              }
72          } catch (final Exception ex) {
73              throw new MessagingException("Unable to get river state", ex);
74          }
75  
76          final RiverState rs = new RiverState();
77          rs.setFolderUrl(folder.getURLName().toString());
78          rs.setLastUid(1L);
79          rs.setExists(true);
80          return rs;
81  
82      }
83  
84      public String index() {
85          return index;
86      }
87  
88      public ElasticsearchRiverStateManager index(final String index) {
89          this.index = index;
90          return this;
91      }
92  
93      @Override
94      public void onError(final String errmsg, final Folder folder, final Exception e) {
95  
96          logger.error("Folder " + folder.getFullName() + " throws an error:" + errmsg + e, e);
97  
98          try {
99              client.prepareIndex(index(), RIVERSTATE_TYPE, ERRORS_ID + "_" + folder.getURLName().toString().hashCode())
100                     .setSource(folder.getURLName().toString(), errmsg + e).execute().actionGet();
101 
102         } catch (final Exception ex) {
103             logger.error("Unable to log an error because of " + ex + errmsg, e);
104         }
105 
106     }
107 
108     @Override
109     public void onError(final String errmsg, final Message msg, final Exception e) {
110 
111         try {
112             logger.error("Message " + ((MimeMessage) msg).getMessageID() + " throws an error: " + errmsg + e, e);
113 
114             client.prepareIndex(index(), RIVERSTATE_TYPE, ERRORS_ID + "_" + ((MimeMessage) msg).getMessageID().hashCode())
115                     .setSource(((MimeMessage) msg).getMessageID(), errmsg + e).execute().actionGet();
116 
117         } catch (final Exception ex) {
118             logger.error("Unable to log an error because of " + ex + errmsg, e);
119         }
120     }
121 
122     @Override
123     public void setRiverState(final RiverState state) throws MessagingException {
124 
125         try {
126             logger.debug("set riverstate " + state);
127 
128             client.prepareIndex(index(), RIVERSTATE_TYPE, FOLDERSTATE_ID + "_" + state.getFolderUrl().hashCode())
129                     .setSource(mapper.writeValueAsString(state)).execute().actionGet();
130 
131             logger.debug("set riverstate done");
132         } catch (final Exception ex) {
133             throw new MessagingException("Unable to set river state", ex);
134         }
135 
136     }
137 
138 }
