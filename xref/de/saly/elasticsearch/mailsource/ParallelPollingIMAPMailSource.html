151     private ProcessResult process(final int messageCount, final int start, final String folderName) {
152 
153         final long startTime = System.currentTimeMillis();
154 
155         //final int netCount = messageCount - (start == 1 ? 0 : start);
156         final int netCount = messageCount - (start == 1 ? 0 : start) + 1; //fixed #15
157 
158         logger.debug("netCount: {}", netCount);
159 
160         if (netCount == 0) {
161             return new ProcessResult(0, 0, 0);
162         }
163         final int block = netCount / threadCount;
164 
165         final List<Future<ProcessResult>> fl = new ArrayList<Future<ProcessResult>>();
166 
167         logger.debug(netCount + "/" + threadCount + "=" + block);
168 
169         int lastStart = 0;
170 
171         for (int i = 1; i <= threadCount; i++) {
172 
173             final int _start = i == 1 ? start : lastStart + block + 1;
174             lastStart = _start;
175 
176             final int _end = Math.min(_start + block, messageCount);
177             if (_end < _start) {
178                 continue;
179             }
180 
181             logger.debug("Schedule: " + _start + " - " + _end);
182 
183             final Future<ProcessResult> f = es.submit(new Callable<ProcessResult>() {
184 
185                 @Override
186                 public ProcessResult call() throws Exception {
187                     return processMessageSlice(_start, _end, folderName);
188                 }
189             });
190 
191             fl.add(f);
192         }
193 
194         long highestUid = 0;
195         int processedCount = 0;
196 
197         for (final Future<ProcessResult> fu : fl) {
198             try {
199                 highestUid = Math.max(highestUid, fu.get().highestUid);
200                 processedCount += fu.get().processedCount;
201                 logger.debug("Finished with " + fu.get());
202             } catch (final Exception e) {
203                 logger.error("Unable to process some mails due to {}", e, e.toString());
204             }
205         }
206 
207         final long endTime = System.currentTimeMillis() + 1;
208 
209         return new ProcessResult(highestUid, processedCount, endTime - startTime);
210 
211     }

213     private ProcessResult processMessageSlice(final int start, final int end, final String folderName) throws Exception {
214 
215         logger.debug("processMessageSlice() started with " + start + "/" + end + "/" + folderName);
216         final long startTime = System.currentTimeMillis();
217         final Store store = Session.getInstance(props).getStore();
218         store.connect(user, password);
219         final Folder folder = store.getFolder(folderName);
220         final UIDFolder uidfolder = (UIDFolder) folder;
221 
222         IMAPUtils.open(folder);
223 
224         try {
225 
226             final Message[] msgs = folder.getMessages(start, end);
227             folder.fetch(msgs, IMAPUtils.FETCH_PROFILE_HEAD);
228 
229             logger.debug("folder fetch done");
230 
231             long highestUid = 0;
232             int processedCount = 0;
233 
234             for (final Message m : msgs) {
235                 try {
236 
237                     IMAPUtils.open(folder);
238                     final long uid = uidfolder.getUID(m);
239 
240                     mailDestination.onMessage(m);
241 
242                     highestUid = Math.max(highestUid, uid);
243                     processedCount++;
244 
245                     if (Thread.currentThread().isInterrupted()) {
246                         break;
247                     }
248 
249                 } catch (final Exception e) {
250                     stateManager.onError("Unable to make indexable message", m, e);
251                     logger.error("Unable to make indexable message due to {}", e, e.toString());
252 
253                     IMAPUtils.open(folder);
254                 }
255             }
256 
257             final long endTime = System.currentTimeMillis() + 1;
258             final ProcessResult pr = new ProcessResult(highestUid, processedCount, endTime - startTime);
259             logger.debug("processMessageSlice() ended with " + pr);
260             return pr;
261 
262         } finally {
263 
264             IMAPUtils.close(folder);
265             IMAPUtils.close(store);
266         }
267 
268     }

270     @SuppressWarnings({ "rawtypes", "unchecked" })
271     protected void fetch(final Folder folder) throws MessagingException, IOException {
272 
273         if ((folder.getType() & Folder.HOLDS_MESSAGES) == 0) {
274             logger.warn("Folder {} cannot hold messages", folder.getFullName());
275             return;
276 
277         }
278 
279         final int messageCount = folder.getMessageCount();
280 
281         final UIDFolder uidfolder = (UIDFolder) folder;
282         final long servervalidity = uidfolder.getUIDValidity();
283         final RiverState riverState = stateManager.getRiverState(folder);
284         final Long localvalidity = riverState.getUidValidity();
285 
286         logger.info("Fetch mails from folder {} ({})", folder.getURLName().toString(), messageCount);
287 
288         logger.debug("Server uid validity: {}, Local uid validity: {}", servervalidity, localvalidity);
289 
290         if (localvalidity == null || localvalidity.longValue() != servervalidity) {
291 
292             logger.debug("UIDValidity fail, full resync " + localvalidity + "!=" + servervalidity);
293 
294             if (localvalidity != null) {
295                 mailDestination.clearDataForFolder(folder.getFullName());
296             }
297 
298             final ProcessResult result = process(messageCount, 1, folder.getFullName());
299 
300             riverState.setLastCount(result.getProcessedCount());
301 
302             if (result.getProcessedCount() > 0) {
303                 riverState.setLastIndexed(new Date());
304             }
305 
306             if (result.getProcessedCount() > 0) {
307                 riverState.setLastTook(result.getTook());
308             }
309 
310             riverState.setLastSchedule(new Date());
311 
312             if (result.getProcessedCount() > 0 && result.getHighestUid() > 0) {
313                 riverState.setLastUid(result.getHighestUid());
314             }
315 
316             riverState.setUidValidity(servervalidity);
317             stateManager.setRiverState(riverState);
318 
319             logger.info("Initiailly processed {} mails for folder {}", result.getProcessedCount(), folder.getFullName());
320             logger.debug("Processed result {}", result.toString());
321 
322         } else {
323 
324             if (messageCount == 0) {
325                 logger.debug("Folder {} is empty", folder.getFullName());
326             } else {
327 
328                 if (withFlagSync) {
329                     // detect flag change
330                     final Message[] flagMessages = folder.getMessages();
331                     folder.fetch(flagMessages, IMAPUtils.FETCH_PROFILE_FLAGS_UID);
332 
333                     for (final Message message : flagMessages) {
334                         try {
335 
336                             final long uid = ((UIDFolder) message.getFolder()).getUID(message);
337 
338                             final String id = uid + "::" + message.getFolder().getURLName();
339 
340                             final int storedHashcode = mailDestination.getFlaghashcode(id);
341 
342                             if (storedHashcode == -1) {
343                                 // New mail which is not indexed yet
344                                 continue;
345                             }
346 
347                             final int flagHashcode = message.getFlags().hashCode();
348 
349                             if (flagHashcode != storedHashcode) {
350                                 // flags change for this message, must update
351                                 mailDestination.onMessage(message);
352 
353                                 if (logger.isDebugEnabled()) {
354                                     logger.debug("Update " + id + " because of flag change");
355                                 }
356                             }
357                         } catch (final Exception e) {
358                             logger.error("Error detecting flagchanges for message " + ((MimeMessage) message).getMessageID(), e);
359                             stateManager.onError("Error detecting flagchanges", message, e);
360                         }
361                     }
362                 }
363 
364                 final long highestUID = riverState.getLastUid(); // this uid is
365                                                                  // already
366                                                                  // processed
367 
368                 logger.debug("highestUID: {}", highestUID);
369 
370                 final Message[] msgsnew = uidfolder.getMessagesByUID(highestUID, UIDFolder.LASTUID);
371 
372                 // msgnew.size is always >= 1
373                 if (highestUID > 0 && uidfolder.getUID(msgsnew[0]) <= highestUID) {
374                     // msgsnew = (Message[]) ArrayUtils.remove(msgsnew, 0);
375                 }
376 
377                 if (msgsnew.length > 0) {
378 
379                     logger.info("{} new messages in folder {}", msgsnew.length, folder.getFullName());
380 
381                     final int start = msgsnew[0].getMessageNumber();
382 
383                     final ProcessResult result = process(messageCount, start, folder.getFullName());
384 
385                     riverState.setLastCount(result.getProcessedCount());
386 
387                     if (result.getProcessedCount() > 0) {
388                         riverState.setLastIndexed(new Date());
389                     }
390 
391                     if (result.getProcessedCount() > 0) {
392                         riverState.setLastTook(result.getTook());
393                     }
394 
395                     riverState.setLastSchedule(new Date());
396 
397                     if (result.getProcessedCount() > 0 && result.getHighestUid() > 0) {
398                         riverState.setLastUid(result.getHighestUid());
399                     }
400 
401                     riverState.setUidValidity(servervalidity);
402                     stateManager.setRiverState(riverState);
403 
404                     logger.info("Not initiailly processed {} mails for folder {}", result.getProcessedCount(), folder.getFullName());
405                     logger.debug("Processed result {}", result.toString());
406                 } else {
407                     logger.debug("no new messages");
408                 }
409 
410             }
411             // check for expunged/deleted messages
412 
413             final Set<Long> serverMailSet = new HashSet<Long>();
414 
415             final long oldmailUid = riverState.getLastUid();
416             logger.debug("oldmailuid {}", oldmailUid);
417 
418             final Message[] msgsold = uidfolder.getMessagesByUID(1, oldmailUid);
419 
420             folder.fetch(msgsold, IMAPUtils.FETCH_PROFILE_UID);
421 
422             for (final Message m : msgsold) {
423                 try {
424                     final long uid = uidfolder.getUID(m);
425                     serverMailSet.add(uid);
426 
427                 } catch (final Exception e) {
428                     stateManager.onError("Unable to handle old message ", m, e);
429                     logger.error("Unable to handle old message due to {}", e, e.toString());
430 
431                     IMAPUtils.open(folder);
432                 }
433             }
434 
435             final Set localMailSet = new HashSet(mailDestination.getCurrentlyStoredMessageUids(folder.getFullName(), false));
436 
437             logger.debug("Check now " + localMailSet.size() + " server mails for expunge");
438 
439             localMailSet.removeAll(serverMailSet);
440             // localMailSet has now the ones that are not on server
441 
442             logger.info(localMailSet.size() + " messages were locally deleted, because they are expunged on server.");
443 
444             mailDestination.onMessageDeletes(localMailSet, folder.getFullName(), false);
445 
446         }
447 
448     }

450     protected void fetch(final Pattern pattern, final String folderName) throws MessagingException, IOException {
451 
452         logger.debug("fetch() - pattern: {}, folderName: {}", pattern, folderName);
453 
454         final Store store = Session.getInstance(props).getStore();
455         store.connect(user, password);
456 
457         try {
458             for (final String fol : mailDestination.getFolderNames()) {
459                 if (store.getFolder(fol).exists()) {
460                     logger.debug("{} exists", fol);
461                 } else {
462                     logger.info("Folder {} does not exist on the server, will remove it (and its content) also locally", fol);
463                     final RiverState riverState = stateManager.getRiverState(store.getFolder(fol));
464                     riverState.setExists(false);
465                     stateManager.setRiverState(riverState);
466 
467                     try {
468                         mailDestination.clearDataForFolder(fol);
469                     } catch (final Exception e) {
470                         stateManager.onError("Unable to clean data for stale folder", store.getFolder(fol), e);
471                     }
472                 }
473             }
474         } catch (final IndexMissingException ime) {
475             logger.debug(ime.toString());
476 
477         } catch (final Exception e) {
478             logger.error("Error checking for stale folders", e);
479         }
480 
481         final boolean isRoot = StringUtils.isEmpty(folderName);
482         final Folder folder = isRoot ? store.getDefaultFolder() : store.getFolder(folderName);
483 
484         try {
485 
486             if (!folder.exists()) {
487                 logger.error("Folder {} does not exist on the server", folder.getFullName());
488                 return;
489             }
490 
491             logger.debug("folderName: {} is root: {}", folderName, isRoot);
492 
493             /*if (logger.isDebugEnabled() && store instanceof IMAPStore) {
494 
495             	IMAPStore imapStore = (IMAPStore) store;
496             	logger.debug("IMAP server capability info");
497             	logger.debug("IMAP4rev1: "
498             			+ imapStore.hasCapability("IMAP4rev1"));
499             	logger.debug("IDLE: " + imapStore.hasCapability("IDLE"));
500             	logger.debug("ID: " + imapStore.hasCapability("ID"));
501             	logger.debug("UIDPLUS: " + imapStore.hasCapability("UIDPLUS"));
502             	logger.debug("CHILDREN: " + imapStore.hasCapability("CHILDREN"));
503             	logger.debug("NAMESPACE: "
504             			+ imapStore.hasCapability("NAMESPACE"));
505             	logger.debug("NOTIFY: " + imapStore.hasCapability("NOTIFY"));
506             	logger.debug("CONDSTORE: "
507             			+ imapStore.hasCapability("CONDSTORE"));
508             	logger.debug("QRESYNC: " + imapStore.hasCapability("QRESYNC"));
509             	logger.debug("STATUS: " + imapStore.hasCapability("STATUS"));
510             }*/
511 
512             if (pattern != null && !isRoot && !pattern.matcher(folder.getFullName()).matches()) {
513                 logger.info(folder.getFullName() + " does not match pattern " + pattern.toString());
514                 return;
515             }
516 
517             IMAPUtils.open(folder);
518 
519             recurseFolders(folder, pattern);
520 
521         } finally {
522 
523             IMAPUtils.close(folder);
524             IMAPUtils.close(store);
525 
526         }
527     }

529     protected void recurseFolders(final Folder folder, final Pattern pattern) throws MessagingException, IOException {
530 
531         if (folder != null) {
532 
533             if (es == null || es.isShutdown() || es.isTerminated() || Thread.currentThread().isInterrupted()) {
534 
535                 logger.warn("Stop processing of mails due to mail source is closed");
536                 return;
537 
538             }
539 
540             if ((folder.getType() & Folder.HOLDS_MESSAGES) != 0) {
541 
542                 if (pattern != null && !pattern.matcher(folder.getFullName()).matches()) {
543                     logger.info("Pattern {} does not match {}", pattern.pattern(), folder.getFullName());
544                     return;
545                 }
546                 IMAPUtils.open(folder);
547 
548                 try {
549                     fetch(folder);
550                 } finally {
551                     IMAPUtils.close(folder);
552                     logger.debug("fetch {} done", folder.getFullName());
553                 }
554             }
555 
556             if ((folder.getType() & Folder.HOLDS_FOLDERS) != 0) {
557                 for (final Folder subfolder : folder.list()) {
558 
559                     recurseFolders(subfolder, pattern);
560 
561                 }
562             }
563 
564         }
565

568     private static class ProcessResult {
569         private long highestUid = 1;
570 
571         private final int processedCount;
572         private final long took;
573 
574         public ProcessResult(final long highestUid, final int processedCount, final long took) {
575             super();
576             this.highestUid = highestUid < 1 ? 1L : highestUid;
577             this.processedCount = processedCount;
578             this.took = took;
579         }
580 
581         public long getHighestUid() {
582             return highestUid;
583         }
584 
585         public int getProcessedCount() {
586             return processedCount;
587         }
588 
589         public long getTook() {
590             return took;
591         }
592 
593         @Override
594         public String toString() {
595             return "ProcessResult [highestUid=" + highestUid + ", processedCount=" + processedCount + ", took=" + took + "]";
596         }
