65      public static IndexableMailMessage fromJavaMailMessage(final Message jmm, final boolean withTextContent, final boolean withHtmlContent, final boolean preferHtmlContent, final boolean withAttachments,
66              final boolean stripTags, List<String> headersToFields) throws MessagingException, IOException {
67          final IndexableMailMessage im = new IndexableMailMessage();
68  
69          @SuppressWarnings("unchecked")
70          final Enumeration<Header> allHeaders = jmm.getAllHeaders();
71  
72          final Set<IndexableHeader> headerList = new HashSet<IndexableHeader>();
73          while (allHeaders.hasMoreElements()) {
74              final Header h = allHeaders.nextElement();
75              headerList.add(new IndexableHeader(h.getName(), h.getValue()));
76          }
77  
78          im.setHeaders(headerList.toArray(new IndexableHeader[headerList.size()]));
79  
80          im.setSelectedHeaders(extractHeaders(im.getHeaders(), headersToFields));
81  
82          if (jmm.getFolder() instanceof POP3Folder) {
83              im.setPopId(((POP3Folder) jmm.getFolder()).getUID(jmm));
84              im.setMailboxType("POP");
85  
86          } else {
87              im.setMailboxType("IMAP");
88          }
89  
90          if (jmm.getFolder() instanceof UIDFolder) {
91              im.setUid(((UIDFolder) jmm.getFolder()).getUID(jmm));
92          }
93  
94          im.setFolderFullName(jmm.getFolder().getFullName());
95  
96          im.setFolderUri(jmm.getFolder().getURLName().toString());
97  
98          im.setContentType(jmm.getContentType());
99          im.setSubject(jmm.getSubject());
100         im.setSize(jmm.getSize());
101         im.setSentDate(jmm.getSentDate());
102 
103         if (jmm.getReceivedDate() != null) {
104             im.setReceivedDate(jmm.getReceivedDate());
105         }
106 
107         if (jmm.getFrom() != null && jmm.getFrom().length > 0) {
108             im.setFrom(Address.fromJavaMailAddress(jmm.getFrom()[0]));
109         }
110 
111         if (jmm.getRecipients(RecipientType.TO) != null) {
112             im.setTo(Address.fromJavaMailAddress(jmm.getRecipients(RecipientType.TO)));
113         }
114 
115         if (jmm.getRecipients(RecipientType.CC) != null) {
116             im.setCc(Address.fromJavaMailAddress(jmm.getRecipients(RecipientType.CC)));
117         }
118 
119         if (jmm.getRecipients(RecipientType.BCC) != null) {
120             im.setBcc(Address.fromJavaMailAddress(jmm.getRecipients(RecipientType.BCC)));
121         }
122 
123         if (withTextContent) {
124 
125             // try {
126 
127             String textContent = getText(jmm, 0, preferHtmlContent);
128 
129             if (stripTags) {
130                 textContent = stripTags(textContent);
131             }
132 
133             im.setTextContent(textContent);
134             // } catch (final Exception e) {
135             // logger.error("Unable to retrieve text content for message {} due to {}",
136             // e, ((MimeMessage) jmm).getMessageID(), e);
137             // }
138         }
139 
140         if (withHtmlContent) { 
141 
142             // try {
143 
144             String htmlContent = getText(jmm, 0, true);
145 
146             im.setHtmlContent(htmlContent);
147             // } catch (final Exception e) {
148             // logger.error("Unable to retrieve text content for message {} due to {}",
149             // e, ((MimeMessage) jmm).getMessageID(), e);
150             // }
151         }
152 
153         if (withAttachments) {
154 
155             try {
156                 final Object content = jmm.getContent();
157 
158                 // look for attachments
159                 if (jmm.isMimeType("multipart/*") && content instanceof Multipart) {
160                     List<ESAttachment> attachments = new ArrayList<ESAttachment>();
161 
162                     final Multipart multipart = (Multipart) content;
163 
164                     for (int i = 0; i < multipart.getCount(); i++) {
165                         final BodyPart bodyPart = multipart.getBodyPart(i);
166                         if (!Part.ATTACHMENT.equalsIgnoreCase(bodyPart.getDisposition()) && !StringUtils.isNotBlank(bodyPart.getFileName())) {
167                             continue; // dealing with attachments only
168                         }
169                         final InputStream is = bodyPart.getInputStream();
170                         final byte[] bytes = IOUtils.toByteArray(is);
171                         IOUtils.closeQuietly(is);
172                         attachments.add(new ESAttachment(bodyPart.getContentType(), bytes, bodyPart.getFileName()));
173                     }
174 
175                     if (!attachments.isEmpty()) {
176                         im.setAttachments(attachments.toArray(new ESAttachment[attachments.size()]));
177                         im.setAttachmentCount(im.getAttachments().length);
178                         attachments.clear();
179                         attachments = null;
180                     }
181 
182                 }
183             } catch (final Exception e) {
184                 logger.error("Error indexing attachments (message will be indexed but without attachments) due to {}", e, e.toString());
185             }
186 
187         }
188 
189         im.setFlags(IMAPUtils.toStringArray(jmm.getFlags()));
190         im.setFlaghashcode(jmm.getFlags().hashCode());
191 
192         return im;
193     }

195     private static Map<String, String> extractHeaders(
196             IndexableHeader[] allHeaders, List<String> headersToFields) {
197 
198         Map<String, String> map = new HashMap<>();
199         for(String headerName : headersToFields) {
200             for(IndexableHeader header : allHeaders) {
201                 // e-mail headers are case insensitive
202                 if(headerName.toLowerCase().equals(header.getName().toLowerCase())) {
203                     String fieldName = "header_" + headerName.replaceAll("[^A-Za-z0-9 ]", "_").toLowerCase();
204                     map.put(fieldName, header.getValue());
205                     break;
206                 }
207             }
208         }
209         return map;
210     }

213     private static String getText(final Part p, int depth, final boolean preferHtmlContent) throws MessagingException, IOException {
214 
215         if (depth >= 100) {
216             throw new IOException("Endless recursion detected ");
217         }
218 
219         // TODO fix encoding for buggy encoding headers
220 
221         if (p.isMimeType("text/*")) {
222 
223             Object content = null;
224             try {
225                 content = p.getContent();
226             } catch (final Exception e) {
227                 logger.error("Unable to index the content of a message due to {}", e.toString());
228                 return null;
229             }
230 
231             if (content instanceof String) {
232                 final String s = (String) p.getContent();
233                 return s;
234 
235             }
236 
237             if (content instanceof InputStream) {
238 
239                 final InputStream in = (InputStream) content;
240                 // TODO guess encoding with
241                 // http://code.google.com/p/juniversalchardet/
242                 final String s = IOUtils.toString(in, "UTF-8");
243                 IOUtils.closeQuietly(in);
244                 return s;
245 
246             }
247 
248             throw new MessagingException("Unknown content class representation: " + content.getClass());
249 
250         }
251 
252         if (p.isMimeType("multipart/alternative")) {
253             // prefer plain text over html text
254             final Multipart mp = (Multipart) p.getContent();
255             String text = null;
256             for (int i = 0; i < mp.getCount(); i++) {
257                 final Part bp = mp.getBodyPart(i);
258                 if (bp.isMimeType("text/html")) {
259                     if (text == null) {
260                         text = getText(bp, ++depth, preferHtmlContent);
261                     }
262 		    if (preferHtmlContent) {
263                         return text;
264                     } else {
265                         continue;
266                     }
267                 } else if (bp.isMimeType("text/plain")) {
268                     final String s = getText(bp, ++depth, preferHtmlContent);
269                     if (s != null && !preferHtmlContent) {
270                         return s;
271                     } else {
272                         continue;
273                     }
274                 } else {
275                     return getText(bp, ++depth, preferHtmlContent);
276                 }
277             }
278             return text;
279         } else if (p.isMimeType("multipart/*")) {
280             final Multipart mp = (Multipart) p.getContent();
281             for (int i = 0; i < mp.getCount(); i++) {
282                 final String s = getText(mp.getBodyPart(i), ++depth, preferHtmlContent);
283                 if (s != null) {
284                     return s;
285                 }
286             }
287         }
288 
289         return null;
290     }

292     private static String stripTags(final String text) {
293         if (text == null) {
294             return null;
295         }
296 
297         return text.replaceAll("\\<.*?\\>", "");
298     }

384         }
385         return true;

584     public static class Address {
585         public static Address fromJavaMailAddress(final javax.mail.Address jma) {
586             final Address a = new Address();
587             final InternetAddress ia = (InternetAddress) jma;
588             a.setEmail(ia.getAddress());
589             a.setPersonal(ia.getPersonal());
590 
591             return a;
592         }
593 
594         public static Address[] fromJavaMailAddress(final javax.mail.Address[] jmas) {
595             final Address[] as = new Address[jmas.length];
596             int i = 0;
597             for (final javax.mail.Address jma : jmas) {
598 
599                 final Address a = new Address();
600                 final InternetAddress ia = (InternetAddress) jma;
601                 a.setEmail(ia.getAddress());
602                 a.setPersonal(ia.getPersonal());
603                 as[i++] = a;
604             }
605 
606             return as;
607         }
