081        @Override
082        public void close() {
083            if (es != null) {
084    
085                logger.debug("Initiate shutdown");
086                es.shutdown();
087                try {
088                    if (es.awaitTermination(10, TimeUnit.SECONDS)) {
089                        logger.debug("Shutdown completed gracefully");
090                    } else {
091                        logger.warn("Shutdown completed not gracefully, timeout elapsed");
092                    }
093                } catch (final InterruptedException e) {
094                    logger.warn("Shutdown completed not gracefully, thread interrupted");
095                }
096            }
097        }

146        private ProcessResult process(final int messageCount, final int start, final String folderName) {
147    
148            final long startTime = System.currentTimeMillis();
149    
150            final int netCount = messageCount - (start == 1 ? 0 : start);
151    
152            logger.debug("netCount: {}", netCount);
153    
154            if (netCount == 0) {
155                return new ProcessResult(0, 0, 0);
156            }
157            final int block = netCount / threadCount;
158    
159            final List<Future<ProcessResult>> fl = new ArrayList<Future<ProcessResult>>();
160    
161            logger.debug(netCount + "/" + threadCount + "=" + block);
162    
163            int lastStart = 0;
164    
165            for (int i = 1; i <= threadCount; i++) {
166    
167                final int _start = i == 1 ? start : lastStart + block + 1;
168                lastStart = _start;
169    
170                final int _end = Math.min(_start + block, messageCount);
171                if (_end < _start) {
172                    continue;
173                }
174    
175                logger.debug("Schedule: " + _start + " - " + _end);
176    
177                final Future<ProcessResult> f = es.submit(new Callable<ProcessResult>() {
178    
179                    @Override
180                    public ProcessResult call() throws Exception {
181                        return processMessageSlice(_start, _end, folderName);
182                    }
183                });
184    
185                fl.add(f);
186            }
187    
188            long highestUid = 0;
189            int processedCount = 0;
190    
191            for (final Future<ProcessResult> fu : fl) {
192                try {
193                    highestUid = Math.max(highestUid, fu.get().highestUid);
194                    processedCount += fu.get().processedCount;
195                    logger.debug("Finished with " + fu.get());
196                } catch (final Exception e) {
197                    logger.error("Unable to process some mails due to {}, will retry ...", e, e.toString());
198                }
199            }
200    
201            final long endTime = System.currentTimeMillis() + 1;
202    
203            return new ProcessResult(highestUid, processedCount, endTime - startTime);
204    
205        }

207        private ProcessResult processMessageSlice(final int start, final int end, final String folderName) throws Exception {
208    
209            logger.debug("processMessageSlice() started with " + start + "/" + end + "/" + folderName);
210            final long startTime = System.currentTimeMillis();
211            final Store store = Session.getInstance(props).getStore();
212            store.connect(user, password);
213            final Folder folder = store.getFolder(folderName);
214            final UIDFolder uidfolder = (UIDFolder) folder;
215    
216            IMAPUtils.open(folder);
217    
218            try {
219    
220                final Message[] msgs = folder.getMessages(start, end);
221                folder.fetch(msgs, IMAPUtils.FETCH_PROFILE_HEAD);
222    
223                long highestUid = 0;
224                int processedCount = 0;
225    
226                for (final Message m : msgs) {
227                    try {
228    
229                        final long uid = uidfolder.getUID(m);
230    
231                        mailDestination.onMessage(m);
232    
233                        highestUid = Math.max(highestUid, uid);
234                        processedCount++;
235                    } catch (final Exception e) {
236                        stateManager.onError("Unable to make indexable message", m, e);
237                        logger.error("Unable to make indexable message due to {}", e, e.toString());
238    
239                        IMAPUtils.open(folder);
240                    }
241                }
242    
243                final long endTime = System.currentTimeMillis() + 1;
244                final ProcessResult pr = new ProcessResult(highestUid, processedCount, endTime - startTime);
245                logger.debug("processMessageSlice() ended with " + pr);
246                return pr;
247    
248            } finally {
249    
250                IMAPUtils.close(folder);
251                IMAPUtils.close(store);
252            }
253    
254        }

256        @SuppressWarnings({ "rawtypes", "unchecked" })
257        protected void fetch(final Folder folder) throws MessagingException, IOException {
258    
259            if ((folder.getType() & Folder.HOLDS_MESSAGES) == 0) {
260                logger.warn("Folder {} cannot hold messages", folder.getFullName());
261                return;
262    
263            }
264    
265            final int messageCount = folder.getMessageCount();
266    
267            final UIDFolder uidfolder = (UIDFolder) folder;
268            final long servervalidity = uidfolder.getUIDValidity();
269            final RiverState riverState = stateManager.getRiverState(folder);
270            final Long localvalidity = riverState.getUidValidity();
271    
272            logger.info("Fetch mails from folder {} ({})", folder.getURLName().toString(), messageCount);
273    
274            logger.debug("Server uid validity: {}, Local uid validity: {}", servervalidity, localvalidity);
275    
276            if (localvalidity == null || localvalidity.longValue() != servervalidity) {
277    
278                logger.debug("UIDValidity fail, full resync " + localvalidity + "!=" + servervalidity);
279    
280                if (localvalidity != null) {
281                    mailDestination.clearDataForFolder(folder.getFullName());
282                }
283    
284                final ProcessResult result = process(messageCount, 1, folder.getFullName());
285    
286                riverState.setLastCount(result.getProcessedCount());
287    
288                if (result.getProcessedCount() > 0) {
289                    riverState.setLastIndexed(new Date());
290                }
291    
292                if (result.getProcessedCount() > 0) {
293                    riverState.setLastTook(result.getTook());
294                }
295    
296                riverState.setLastSchedule(new Date());
297    
298                if (result.getProcessedCount() > 0 && result.getHighestUid() > 0) {
299                    riverState.setLastUid(result.getHighestUid());
300                }
301    
302                riverState.setUidValidity(servervalidity);
303                stateManager.setRiverState(riverState);
304    
305                logger.info("Initiailly processed {} mails for folder {}", result.getProcessedCount(), folder.getFullName());
306                logger.debug("Processed result {}", result.toString());
307    
308            } else {
309    
310                if (messageCount == 0) {
311                    logger.debug("Folder {} is empty", folder.getFullName());
312                } else {
313    
314                    final long highestUID = riverState.getLastUid(); // this uid is
315                                                                     // already
316                                                                     // processed
317                                                                     // (except
318                                                                     // highestUID
319                                                                     // == 1, then
320                                                                     // we cannot be
321                                                                     // sure)
322    
323                    logger.debug("highestUID: {}", highestUID);
324    
325                    Message[] msgsnew = uidfolder.getMessagesByUID(highestUID, UIDFolder.LASTUID);
326    
327                    // msgnew.size is always >= 1
328                    if (highestUID > 1 && uidfolder.getUID(msgsnew[0]) <= highestUID) {
329                        msgsnew = (Message[]) ArrayUtils.remove(msgsnew, 0);
330                    }
331    
332                    if (msgsnew.length > 0) {
333    
334                        logger.info("{} new messages in folder {}", msgsnew.length, folder.getFullName());
335    
336                        final int start = msgsnew[0].getMessageNumber();
337    
338                        final ProcessResult result = process(messageCount, start, folder.getFullName());
339    
340                        riverState.setLastCount(result.getProcessedCount());
341    
342                        if (result.getProcessedCount() > 0) {
343                            riverState.setLastIndexed(new Date());
344                        }
345    
346                        if (result.getProcessedCount() > 0) {
347                            riverState.setLastTook(result.getTook());
348                        }
349    
350                        riverState.setLastSchedule(new Date());
351    
352                        if (result.getProcessedCount() > 0 && result.getHighestUid() > 0) {
353                            riverState.setLastUid(result.getHighestUid());
354                        }
355    
356                        riverState.setUidValidity(servervalidity);
357                        stateManager.setRiverState(riverState);
358    
359                        logger.info("Not initiailly processed {} mails for folder {}", result.getProcessedCount(), folder.getFullName());
360                        logger.debug("Processed result {}", result.toString());
361                    } else {
362                        logger.debug("no new messages");
363                    }
364    
365                }
366                // check for expunged/deleted messages
367    
368                final Set<Long> serverMailSet = new HashSet<Long>();
369    
370                final long oldmailUid = riverState.getLastUid();
371                logger.debug("oldmailuid {}", oldmailUid);
372    
373                final Message[] msgsold = uidfolder.getMessagesByUID(1, oldmailUid);
374    
375                folder.fetch(msgsold, IMAPUtils.FETCH_PROFILE_UID);
376    
377                for (final Message m : msgsold) {
378                    try {
379                        final long uid = uidfolder.getUID(m);
380                        serverMailSet.add(uid);
381    
382                    } catch (final Exception e) {
383                        stateManager.onError("Unable to handle old message ", m, e);
384                        logger.error("Unable to handle old message due to {}", e, e.toString());
385    
386                        IMAPUtils.open(folder);
387                    }
388                }
389    
390                final Set localMailSet = new HashSet(mailDestination.getCurrentlyStoredMessageUids(folder.getFullName(), false));
391    
392                logger.debug("Check now " + localMailSet.size() + " server mails for expunge");
393    
394                localMailSet.removeAll(serverMailSet);
395                // localMailSet has now the ones that are not on server
396    
397                logger.info(localMailSet.size() + " messages were locally deleted, because they are expunged on server.");
398    
399                mailDestination.onMessageDeletes(localMailSet, folder.getFullName(), false);
400    
401            }
402    
403        }

405        protected void fetch(final Pattern pattern, final String folderName) throws MessagingException, IOException {
406    
407            logger.debug("fetch() - pattern: {}, folderName: {}", pattern, folderName);
408    
409            final Store store = Session.getInstance(props).getStore();
410            store.connect(user, password);
411    
412            try {
413                for (final String fol : mailDestination.getFolderNames()) {
414                    if (store.getFolder(fol).exists()) {
415                        logger.debug("{} exists", fol);
416                    } else {
417                        logger.info("Folder {} does not exist on the server, will remove it (and its content) also locally", fol);
418                        final RiverState riverState = stateManager.getRiverState(store.getFolder(fol));
419                        riverState.setExists(false);
420                        stateManager.setRiverState(riverState);
421    
422                        try {
423                            mailDestination.clearDataForFolder(fol);
424                        } catch (final Exception e) {
425                            stateManager.onError("Unable to clean data for stale folder", store.getFolder(fol), e);
426                        }
427                    }
428                }
429            } catch (final IndexMissingException ime) {
430                logger.debug(ime.toString());
431    
432            } catch (final Exception e) {
433                logger.error("Error checking for stale folders", e);
434            }
435    
436            final boolean isRoot = StringUtils.isEmpty(folderName);
437            final Folder folder = isRoot ? store.getDefaultFolder() : store.getFolder(folderName);
438    
439            try {
440    
441                if (!folder.exists()) {
442                    logger.error("Folder {} does not exist on the server", folder.getFullName());
443                    return;
444                }
445    
446                logger.debug("folderName: {} is root: {}", folderName, isRoot);
447    
448                /*if (logger.isDebugEnabled() && store instanceof IMAPStore) {
449    
450                    IMAPStore imapStore = (IMAPStore) store;
451                    logger.debug("IMAP server capability info");
452                    logger.debug("IMAP4rev1: "
453                                    + imapStore.hasCapability("IMAP4rev1"));
454                    logger.debug("IDLE: " + imapStore.hasCapability("IDLE"));
455                    logger.debug("ID: " + imapStore.hasCapability("ID"));
456                    logger.debug("UIDPLUS: " + imapStore.hasCapability("UIDPLUS"));
457                    logger.debug("CHILDREN: " + imapStore.hasCapability("CHILDREN"));
458                    logger.debug("NAMESPACE: "
459                                    + imapStore.hasCapability("NAMESPACE"));
460                    logger.debug("NOTIFY: " + imapStore.hasCapability("NOTIFY"));
461                    logger.debug("CONDSTORE: "
462                                    + imapStore.hasCapability("CONDSTORE"));
463                    logger.debug("QRESYNC: " + imapStore.hasCapability("QRESYNC"));
464                    logger.debug("STATUS: " + imapStore.hasCapability("STATUS"));
465                }*/
466    
467                if (pattern != null && !isRoot && !pattern.matcher(folder.getFullName()).matches()) {
468                    logger.info(folder.getFullName() + " does not match pattern " + pattern.toString());
469                    return;
470                }
471    
472                IMAPUtils.open(folder);
473    
474                recurseFolders(folder, pattern);
475    
476            } finally {
477    
478                IMAPUtils.close(folder);
479                IMAPUtils.close(store);
480    
481            }
482        }

484        protected void recurseFolders(final Folder folder, final Pattern pattern) throws MessagingException, IOException {
485    
486            if (folder != null) {
487    
488                if (es == null || es.isShutdown() || es.isTerminated()) {
489    
490                    logger.warn("Stop processing of mails due to mail source is closed");
491                    return;
492    
493                }
494    
495                if ((folder.getType() & Folder.HOLDS_MESSAGES) != 0) {
496    
497                    if (pattern != null && !pattern.matcher(folder.getFullName()).matches()) {
498                        logger.info("Pattern {} does not match {}", pattern.pattern(), folder.getFullName());
499                        return;
500                    }
501                    IMAPUtils.open(folder);
502                    try {
503                        fetch(folder);
504                    } finally {
505                        IMAPUtils.close(folder);
506                        logger.debug("fetch {} done", folder.getFullName());
507                    }
508                }
509    
510                if ((folder.getType() & Folder.HOLDS_FOLDERS) != 0) {
511                    for (final Folder subfolder : folder.list()) {
512    
513                        recurseFolders(subfolder, pattern);
514    
515                    }
516                }
517    
518            }
519    
520        }

522        private static class ProcessResult {
523            private long highestUid = 1;
524    
525            private final int processedCount;
526            private final long took;
527    
528            public ProcessResult(final long highestUid, final int processedCount, final long took) {
529                super();
530                this.highestUid = highestUid < 1 ? 1L : highestUid;
531                this.processedCount = processedCount;
532                this.took = took;
533            }
534    
535            public long getHighestUid() {
536                return highestUid;
537            }
538    
539            public int getProcessedCount() {
540                return processedCount;
541            }
542    
543            public long getTook() {
544                return took;
545            }
546    
547            @Override
548            public String toString() {
549                return "ProcessResult [highestUid=" + highestUid + ", processedCount=" + processedCount + ", took=" + took + "]";
550            }
551    
552        }
