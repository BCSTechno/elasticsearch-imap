145    private ProcessResult process(final int messageCount, final int start, final String folderName) {
146
147        final long startTime = System.currentTimeMillis();
148
149        final int netCount = messageCount - (start == 1 ? 0 : start);
150
151        logger.debug("netCount: {}", netCount);
152
153        if (netCount == 0) {
154            return new ProcessResult(0, 0);
155        }
156        final int block = netCount / threadCount;
157
158        final List<Future<ProcessResult>> fl = new ArrayList<Future<ProcessResult>>();
159
160        logger.debug(netCount + "/" + threadCount + "=" + block);
161
162        int lastStart = 0;
163
164        for (int i = 1; i <= threadCount; i++) {
165
166            final int _start = i == 1 ? start : lastStart + block + 1;
167            lastStart = _start;
168
169            final int _end = Math.min(_start + block, messageCount);
170            if (_end < _start) {
171                continue;
172            }
173
174            logger.debug("Schedule: " + _start + " - " + _end);
175
176            final Future<ProcessResult> f = es.submit(new Callable<ProcessResult>() {
177
178                @Override
179                public ProcessResult call() throws Exception {
180                    return processMessageSlice(_start, _end, folderName);
181                }
182            });
183
184            fl.add(f);
185        }
186
187        int processedCount = 0;
188
189        for (final Future<ProcessResult> fu : fl) {
190            try {
191
192                processedCount += fu.get().processedCount;
193            } catch (final Exception e) {
194                logger.error("Unable to process some mails due to {}", e, e.toString());
195            }
196        }
197
198        final long endTime = System.currentTimeMillis() + 1;
199
200        return new ProcessResult(processedCount, endTime - startTime);
201
202    }

204    private ProcessResult processMessageSlice(final int start, final int end, final String folderName) throws Exception {
205
206        final long startTime = System.currentTimeMillis();
207        final Store store = Session.getInstance(props).getStore();
208        store.connect(user, password);
209        final Folder folder = store.getFolder(folderName);
210
211        IMAPUtils.open(folder);
212
213        try {
214
215            final Message[] msgs = folder.getMessages(start, end);
216            folder.fetch(msgs, IMAPUtils.FETCH_PROFILE_HEAD);
217
218            int processedCount = 0;
219
220            for (final Message m : msgs) {
221                try {
222                    mailDestination.onMessage(m);
223                    processedCount++;
224
225                    if (Thread.currentThread().isInterrupted()) {
226                        break;
227                    }
228
229                } catch (final Exception e) {
230                    stateManager.onError("Unable to make indexable message", m, e);
231                    logger.error("Unable to make indexable message due to {}", e, e.toString());
232
233                    IMAPUtils.open(folder);
234                }
235            }
236
237            final long endTime = System.currentTimeMillis() + 1;
238            return new ProcessResult(processedCount, endTime - startTime);
239
240        } finally {
241
242            IMAPUtils.close(folder);
243            IMAPUtils.close(store);
244        }
245
246    }

248    @SuppressWarnings({ "rawtypes", "unchecked" })
249    protected void fetch(final Folder folder) throws MessagingException, IOException {
250
251        if ((folder.getType() & Folder.HOLDS_MESSAGES) == 0) {
252            logger.warn("Folder {} cannot hold messages", folder.getFullName());
253            return;
254
255        }
256
257        final int messageCount = folder.getMessageCount(); // can change during
258                                                           // method
259        final RiverState riverState = stateManager.getRiverState(folder);
260
261        logger.info("Fetch mails from folder {} ({})", folder.getURLName().toString(), messageCount);
262
263        final Map<String, Integer> serverMailSet = new HashMap<String, Integer>(messageCount);// can
264                                                                                              // become
265                                                                                              // stale
266        int highestMsgNum = 1;
267        Set localMailSet = new HashSet();
268        try {
269            localMailSet = new HashSet(mailDestination.getCurrentlyStoredMessageUids(folder.getFullName(), true)); // will
270                                                                                                                   // not
271                                                                                                                   // change
272                                                                                                                   // during
273                                                                                                                   // this
274                                                                                                                   // method
275        } catch (final IndexMissingException ime) {
276            logger.debug(ime.toString());
277
278        }
279
280        if (messageCount > 0) {
281
282            // get all uids from server
283            final Message[] allmsg = folder.getMessages();
284            folder.fetch(allmsg, IMAPUtils.FETCH_PROFILE_UID);
285
286            for (final Message m : allmsg) {
287                try {
288                    final String uid = ((POP3Folder) folder).getUID(m);
289                    serverMailSet.put(uid, m.getMessageNumber());
290
291                } catch (final Exception e) {
292                    stateManager.onError("Unable to handle message ", m, e);
293                    logger.error("Unable to handle message due to {}", e, e.toString());
294
295                    IMAPUtils.open(folder);
296                }
297            }
298
299            try {
300
301                // get all local store message uids
302                // only retrieve the ones which are not stored locally
303                // loop through local mails and look for highest msg num
304                for (final Object retuid : localMailSet) {
305
306                    final Integer tmpNum = serverMailSet.get(retuid);
307
308                    if (tmpNum == null) {
309                        // message was deleted on server
310                        // handle later
311                    } else {
312                        // evaluate highest msg num (not id)
313                        highestMsgNum = Math.max(highestMsgNum, serverMailSet.get(retuid).intValue());
314
315                    }
316
317                }
318
319                logger.debug("highestMsgNum: {}", highestMsgNum);
320            } catch (final Exception e) {
321                logger.error("Error evalutating new messages. Will download all ... due to {}", e, e.toString());
322            }
323
324            final ProcessResult result = process(messageCount, highestMsgNum, folder.getFullName());
325
326            riverState.setLastCount(result.getProcessedCount());
327
328            if (result.getProcessedCount() > 0) {
329                riverState.setLastIndexed(new Date());
330            }
331
332            riverState.setLastTook(result.getTook());
333            riverState.setLastSchedule(new Date());
334            stateManager.setRiverState(riverState);
335
336            logger.info("Processed {} mails", result.getProcessedCount());
337        } else {
338            logger.debug("Mailbox empty");
339        }
340
341        // check for expunged/deleted messages
342        logger.debug("Check now " + localMailSet.size() + " mails for expunge");
343
344        localMailSet.removeAll(serverMailSet.keySet());
345        // tmpset has now the ones that are not on server
346
347        logger.info(localMailSet.size() + " messages were locally deleted, because they are expunged on server.");
348
349        mailDestination.onMessageDeletes(localMailSet, folder.getFullName(), true);
350
351    }

353    protected void fetch(final Pattern unusedPattern, final String unusedFolderName) throws MessagingException, IOException {
354
355        logger.debug("fetch() - folderName: {}", "INBOX");
356
357        final Store store = Session.getInstance(props).getStore();
358        store.connect(user, password);
359
360        final Folder folder = store.getDefaultFolder();
361        try {
362
363            if (!folder.exists()) {
364                logger.error("Folder {} does not exist on the server", folder.getFullName());
365                return;
366            }
367
368            IMAPUtils.open(folder);
369
370            recurseFolders(folder, null);
371
372        } finally {
373
374            IMAPUtils.close(folder);
375            IMAPUtils.close(store);
376        }
377    }
378
379    protected void recurseFolders(final Folder folder, final Pattern pattern) throws MessagingException, IOException {
380
381        if (folder != null) {
382
383            if (es == null || es.isShutdown() || es.isTerminated() || Thread.currentThread().isInterrupted()) {
384
385                logger.warn("Stop processing of mails due to mail source is closed");
386                return;
387
388            }
389
390            if ((folder.getType() & Folder.HOLDS_MESSAGES) != 0) {
391
392                if (pattern != null && !pattern.matcher(folder.getFullName()).matches()) {
393                    logger.trace("Pattern {} does not match {}", pattern.pattern(), folder.getFullName());
394                    return;
395                }
396                IMAPUtils.open(folder);
397                try {
398                    fetch(folder);
399                } finally {
400                    IMAPUtils.close(folder);
401                }
402            }
403
404            if ((folder.getType() & Folder.HOLDS_FOLDERS) != 0) {
405                for (final Folder subfolder : folder.list()) {
406
407                    recurseFolders(subfolder, pattern);
408
409                }
410            }
411
412        }
413
