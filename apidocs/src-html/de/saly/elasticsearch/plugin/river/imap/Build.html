026package de.saly.elasticsearch.plugin.river.imap;
027
028import java.io.ByteArrayOutputStream;
029import java.io.InputStream;
030import java.io.StringReader;
031import java.net.URL;
032import java.util.Enumeration;
033import java.util.Properties;
034
035//copied from https://github.com/jprante/elasticsearch-river-jdbc/blob/master/src/main/java/org/xbib/elasticsearch/plugin/river/jdbc/Build.java
036public class Build {
037
038    private static final Build INSTANCE;
039
040    static {
041        String version = "NA";
042        String hash = "NA";
043        String hashShort = "NA";
044        String timestamp = "NA";
045        String date = "NA";
046
047        try {
048            final String pluginName = IMAPRiverPlugin.class.getName();
049            final Enumeration<URL> e = IMAPRiverPlugin.class.getClassLoader().getResources("es-plugin.properties");
050            while (e.hasMoreElements()) {
051                final URL url = e.nextElement();
052                final InputStream in = url.openStream();
053                final ByteArrayOutputStream out = new ByteArrayOutputStream();
054                final byte[] buffer = new byte[1024];
055                int len;
056                while ((len = in.read(buffer)) != -1) {
057                    out.write(buffer, 0, len);
058                }
059                in.close();
060                final Properties props = new Properties();
061                props.load(new StringReader(new String(out.toByteArray())));
062                final String plugin = props.getProperty("plugin");
063                if (pluginName.equals(plugin)) {
064                    version = props.getProperty("version");
065                    hash = props.getProperty("hash");
066                    if (!"NA".equals(hash)) {
067                        hashShort = hash.substring(0, 7);
068                    }
069                    timestamp = props.getProperty("timestamp");
070                    date = props.getProperty("date");
071                }
072            }
073        } catch (final Throwable e) {
074            // just ignore...
075        }
076        INSTANCE = new Build(version, hash, hashShort, timestamp, date);
077    }
078
079    public static Build getInstance() {
080        return INSTANCE;
081    }
082
083    private final String date;
084
085    private final String hash;
086
087    private final String hashShort;
088
089    private final String timestamp;
090
091    private final String version;
092
093    Build(final String version, final String hash, final String hashShort, final String timestamp, final String date) {
094        this.version = version;
095        this.hash = hash;
096        this.hashShort = hashShort;
097        this.timestamp = timestamp;
098        this.date = date;
099    }
100
101    public String getDate() {
102        return date;
103    }
104
105    public String getHash() {
106        return hash;
107    }
108
109    public String getShortHash() {
110        return hashShort;
111    }
112
113    public String getTimestamp() {
114        return timestamp;
115    }
116
117    public String getVersion() {
118        return version;
119    }
120
121}
