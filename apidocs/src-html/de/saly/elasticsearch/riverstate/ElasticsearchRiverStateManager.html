026    package de.saly.elasticsearch.riverstate;
027    
028    import javax.mail.Folder;
029    import javax.mail.Message;
030    import javax.mail.MessagingException;
031    import javax.mail.internet.MimeMessage;
032    
033    import org.codehaus.jackson.map.ObjectMapper;
034    import org.codehaus.jackson.type.TypeReference;
035    import org.elasticsearch.action.get.GetResponse;
036    import org.elasticsearch.client.Client;
037    import org.elasticsearch.common.logging.ESLogger;
038    import org.elasticsearch.common.logging.ESLoggerFactory;
039    
040    public class ElasticsearchRiverStateManager implements RiverStateManager {
041    
042        private static final String ERRORS_ID = "errors";
043        private static final String FOLDERSTATE_ID = "folderstate";
044        private static final String RIVERSTATE_TYPE = "imapriverstate";
045        private Client client;
046        private String index;
047        private final ObjectMapper mapper = new ObjectMapper();
048        protected final ESLogger logger = ESLoggerFactory.getLogger(this.getClass().getName());
049    
050        public ElasticsearchRiverStateManager client(final Client client) {
051            this.client = client;
052            return this;
053        }
054    
055        @Override
056        public synchronized RiverState getRiverState(final Folder folder) throws MessagingException {
057    
058            try {
059    
060                if (client.admin().indices().prepareExists(index()).execute().actionGet().isExists()) {
061    
062                    final GetResponse response = client
063                            .prepareGet(index(), RIVERSTATE_TYPE, FOLDERSTATE_ID + "_" + folder.getURLName().toString().hashCode()).execute()
064                            .get();
065    
066                    if (!response.isSourceEmpty()) {
067                        return mapper.readValue(response.getSourceAsString(), new TypeReference<RiverState>() {
068                        });
069    
070                    }
071                }
072            } catch (final Exception ex) {
073                throw new MessagingException("Unable to get river state", ex);
074            }
075    
076            final RiverState rs = new RiverState();
077            rs.setFolderUrl(folder.getURLName().toString());
078            rs.setLastUid(1L);
079            rs.setExists(true);
080            return rs;
081    
082        }
083    
084        public String index() {
085            return index;
086        }
087    
088        public ElasticsearchRiverStateManager index(final String index) {
089            this.index = index;
090            return this;
091        }
092    
093        @Override
094        public void onError(final String errmsg, final Folder folder, final Exception e) {
095    
096            logger.error("Folder " + folder.getFullName() + " throws an error:" + errmsg + e, e);
097    
098            try {
099                client.prepareIndex(index(), RIVERSTATE_TYPE, ERRORS_ID + "_" + folder.getURLName().toString().hashCode())
100                        .setSource(folder.getURLName().toString(), errmsg + e).execute().actionGet();
101    
102            } catch (final Exception ex) {
103                logger.error("Unable to log an error because of " + ex + errmsg, e);
104            }
105    
106        }
107    
108        @Override
109        public void onError(final String errmsg, final Message msg, final Exception e) {
110    
111            try {
112                logger.error("Message " + ((MimeMessage) msg).getMessageID() + " throws an error: " + errmsg + e, e);
113    
114                client.prepareIndex(index(), RIVERSTATE_TYPE, ERRORS_ID + "_" + ((MimeMessage) msg).getMessageID().hashCode())
115                        .setSource(((MimeMessage) msg).getMessageID(), errmsg + e).execute().actionGet();
116    
117            } catch (final Exception ex) {
118                logger.error("Unable to log an error because of " + ex + errmsg, e);
119            }
120        }
121    
122        @Override
123        public void setRiverState(final RiverState state) throws MessagingException {
124    
125            try {
126                logger.debug("set riverstate " + state);
127    
128                client.prepareIndex(index(), RIVERSTATE_TYPE, FOLDERSTATE_ID + "_" + state.getFolderUrl().hashCode())
129                        .setSource(mapper.writeValueAsString(state)).execute().actionGet();
130    
131                logger.debug("set riverstate done");
132            } catch (final Exception ex) {
133                throw new MessagingException("Unable to set river state", ex);
134            }
135    
136        }
137    
138    }
