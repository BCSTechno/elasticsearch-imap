065        public static IndexableMailMessage fromJavaMailMessage(final Message jmm, final boolean withTextContent, final boolean withAttachments,
066                final boolean stripTags) throws MessagingException, IOException {
067            final IndexableMailMessage im = new IndexableMailMessage();
068    
069            @SuppressWarnings("unchecked")
070            final Enumeration<Header> allHeaders = jmm.getAllHeaders();
071    
072            final Map<String, String> headersMap = new HashMap<String, String>();
073            while (allHeaders.hasMoreElements()) {
074                final Header h = allHeaders.nextElement();
075                headersMap.put(h.getName(), h.getValue());
076            }
077    
078            if (jmm.getFolder() instanceof POP3Folder) {
079                im.setPopId(((POP3Folder) jmm.getFolder()).getUID(jmm));
080                im.setMailboxType("POP");
081    
082            } else {
083                im.setMailboxType("IMAP");
084            }
085    
086            im.setHeaders(headersMap);
087    
088            if (jmm.getFolder() instanceof UIDFolder) {
089                im.setUid(((UIDFolder) jmm.getFolder()).getUID(jmm));
090            }
091    
092            im.setFolderFullName(jmm.getFolder().getFullName());
093    
094            im.setFolderUri(jmm.getFolder().getURLName().toString());
095    
096            im.setContentType(jmm.getContentType());
097            im.setSubject(jmm.getSubject());
098            im.setSize(jmm.getSize());
099            im.setSentDate(jmm.getSentDate());
100    
101            if (jmm.getReceivedDate() != null) {
102                im.setReceivedDate(jmm.getReceivedDate());
103            }
104    
105            if (jmm.getFrom() != null && jmm.getFrom().length > 0) {
106                im.setFrom(Address.fromJavaMailAddress(jmm.getFrom()[0]));
107            }
108    
109            if (jmm.getRecipients(RecipientType.TO) != null) {
110                im.setTo(Address.fromJavaMailAddress(jmm.getRecipients(RecipientType.TO)));
111            }
112    
113            if (jmm.getRecipients(RecipientType.CC) != null) {
114                im.setCc(Address.fromJavaMailAddress(jmm.getRecipients(RecipientType.CC)));
115            }
116    
117            if (jmm.getRecipients(RecipientType.BCC) != null) {
118                im.setBcc(Address.fromJavaMailAddress(jmm.getRecipients(RecipientType.BCC)));
119            }
120    
121            if (withTextContent) {
122    
123                try {
124    
125                    String textContent = getText(jmm, 0);
126    
127                    if (stripTags) {
128                        textContent = stripTags(textContent);
129                    }
130    
131                    im.setTextContent(textContent);
132                } catch (final Exception e) {
133                    logger.error("Unable to retrieve text content for message {} due to {}", e, ((MimeMessage) jmm).getMessageID(), e);
134                }
135            }
136    
137            if (withAttachments) {
138    
139                try {
140                    final Object content = jmm.getContent();
141    
142                    // look for attachments
143                    if (jmm.isMimeType("multipart/*") && content instanceof Multipart) {
144                        List<ESAttachment> attachments = new ArrayList<ESAttachment>();
145    
146                        final Multipart multipart = (Multipart) content;
147    
148                        for (int i = 0; i < multipart.getCount(); i++) {
149                            final BodyPart bodyPart = multipart.getBodyPart(i);
150                            if (!Part.ATTACHMENT.equalsIgnoreCase(bodyPart.getDisposition()) && !StringUtils.isNotBlank(bodyPart.getFileName())) {
151                                continue; // dealing with attachments only
152                            }
153                            final InputStream is = bodyPart.getInputStream();
154                            final byte[] bytes = IOUtils.toByteArray(is);
155                            IOUtils.closeQuietly(is);
156                            attachments.add(new ESAttachment(bodyPart.getContentType(), bytes, bodyPart.getFileName()));
157                        }
158    
159                        if (!attachments.isEmpty()) {
160                            im.setAttachments(attachments.toArray(new ESAttachment[attachments.size()]));
161                            im.setAttachmentCount(im.getAttachments().length);
162                            attachments.clear();
163                            attachments = null;
164                        }
165    
166                    }
167                } catch (final Exception e) {
168                    logger.error("Error indexing attachments (message will be indexed but without attachments) due to {}", e, e.toString());
169                }
170    
171            }
172            return im;
173        }

175        private static String getText(final Part p, int depth) throws MessagingException, IOException {
176    
177            if (depth >= 100) {
178                throw new IOException("Endless recursion detected ");
179            }
180    
181            // TODO fix encoding for buggy encoding headers
182            /*
183            String xEncoding[] = p.getHeader("Content-Transfer-Encoding");
184            
185            
186            if(xEncoding != null){
187                    if(xEncoding.length > 0){
188                                                    
189                            if(xEncoding[0].equalsIgnoreCase("7Bit;")){
190                                    
191                                    p.removeHeader("Content-Transfer-Encoding");
192                                    p.addHeader("Content-Transfer-Encoding", "7Bit");
193                            }
194                    }
195                    
196            }*/
197    
198            if (p.isMimeType("text/*")) {
199    
200                final Object content = p.getContent();
201    
202                if (content instanceof String) {
203                    final String s = (String) p.getContent();
204                    return s;
205    
206                }
207    
208                if (content instanceof InputStream) {
209    
210                    final IMAPInputStream in = (IMAPInputStream) p.getContent();
211                    final String s = IOUtils.toString(in, Charset.forName("UTF-8"));
212                    IOUtils.closeQuietly(in);
213                    return s;
214    
215                }
216    
217                throw new MessagingException("unknown content type " + content.getClass());
218    
219            }
220    
221            if (p.isMimeType("multipart/alternative")) {
222                // prefer plain text over html text
223                final Multipart mp = (Multipart) p.getContent();
224                String text = null;
225                for (int i = 0; i < mp.getCount(); i++) {
226                    final Part bp = mp.getBodyPart(i);
227                    if (bp.isMimeType("text/html")) {
228                        if (text == null) {
229                            text = getText(bp, ++depth);
230                        }
231                        continue;
232                    } else if (bp.isMimeType("text/plain")) {
233                        final String s = getText(bp, ++depth);
234                        if (s != null) {
235                            return s;
236                        }
237                    } else {
238                        return getText(bp, ++depth);
239                    }
240                }
241                return text;
242            } else if (p.isMimeType("multipart/*")) {
243                final Multipart mp = (Multipart) p.getContent();
244                for (int i = 0; i < mp.getCount(); i++) {
245                    final String s = getText(mp.getBodyPart(i), ++depth);
246                    if (s != null) {
247                        return s;
248                    }
249                }
250            }
251    
252            return null;
253        }
254    
255        private static String stripTags(final String text) {
256            if (text == null) {
257                return null;
258            }
259    
260            return text.replaceAll("\\<.*?\\>", "");
261        }

339            }
340            return true;

505        public static class Address {
506            public static Address fromJavaMailAddress(final javax.mail.Address jma) {
507                final Address a = new Address();
508                final InternetAddress ia = (InternetAddress) jma;
509                a.setEmail(ia.getAddress());
510                a.setPersonal(ia.getPersonal());
511    
512                return a;
513            }
514    
515            public static Address[] fromJavaMailAddress(final javax.mail.Address[] jmas) {
516                final Address[] as = new Address[jmas.length];
517                int i = 0;
518                for (final javax.mail.Address jma : jmas) {
519    
520                    final Address a = new Address();
521                    final InternetAddress ia = (InternetAddress) jma;
522                    a.setEmail(ia.getAddress());
523                    a.setPersonal(ia.getPersonal());
524                    as[i++] = a;
525                }
526    
527                return as;
528            }
