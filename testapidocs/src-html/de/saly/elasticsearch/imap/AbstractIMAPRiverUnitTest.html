133        protected void checkStoreForTestConnection(final Store store) {
134            if (!store.isConnected()) {
135                IMAPUtils.close(store);
136                throw new RuntimeException("Store not connected");
137            }
138    
139            if (!store.getURLName().getUsername().toLowerCase().startsWith("es_imapriver_unittest")) {
140                IMAPUtils.close(store);
141                throw new RuntimeException("User " + store.getURLName().getUsername() + " belongs not to a valid test mail connection");
142            }
143        }
144    
145        protected void createInitialIMAPTestdata(final Properties props, final String user, final String password, final int count,
146                final boolean deleteall) throws MessagingException {
147            final Session session = Session.getInstance(props);
148            final Store store = session.getStore();
149            store.connect(user, password);
150            checkStoreForTestConnection(store);
151            final Folder root = store.getDefaultFolder();
152            final Folder testroot = root.getFolder("ES-IMAP-RIVER-TESTS");
153            final Folder testrootl2 = testroot.getFolder("Level(2!");
154    
155            if (deleteall) {
156    
157                deleteMailsFromUserMailbox(props, "INBOX", 0, -1, user, password);
158    
159                if (testroot.exists()) {
160                    testroot.delete(true);
161                }
162    
163                final Folder testrootenamed = root.getFolder("renamed_from_ES-IMAP-RIVER-TESTS");
164                if (testrootenamed.exists()) {
165                    testrootenamed.delete(true);
166                }
167    
168            }
169    
170            if (!testroot.exists()) {
171    
172                testroot.create(Folder.HOLDS_FOLDERS & Folder.HOLDS_MESSAGES);
173                testroot.open(Folder.READ_WRITE);
174    
175                testrootl2.create(Folder.HOLDS_FOLDERS & Folder.HOLDS_MESSAGES);
176                testrootl2.open(Folder.READ_WRITE);
177    
178            }
179    
180            final Folder inbox = root.getFolder("INBOX");
181            inbox.open(Folder.READ_WRITE);
182    
183            final Message[] msgs = new Message[count];
184    
185            for (int i = 0; i < count; i++) {
186                final MimeMessage message = new MimeMessage(session);
187                message.setFrom(new InternetAddress(EMAIL_TO));
188                message.addRecipient(Message.RecipientType.TO, new InternetAddress(EMAIL_USER_ADDRESS));
189                message.setSubject(EMAIL_SUBJECT + "::" + i);
190                message.setText(EMAIL_TEXT + "::" + SID++);
191                message.setSentDate(new Date());
192                msgs[i] = message;
193    
194            }
195    
196            inbox.appendMessages(msgs);
197            testroot.appendMessages(msgs);
198            testrootl2.appendMessages(msgs);
199    
200            IMAPUtils.close(inbox);
201            IMAPUtils.close(testrootl2);
202            IMAPUtils.close(testroot);
203            IMAPUtils.close(store);
204    
205        }

207        protected void deleteMailsFromUserMailbox(final Properties props, final String folderName, final int start, final int deleteCount,
208                final String user, final String password) throws MessagingException {
209            final Store store = Session.getInstance(props).getStore();
210    
211            store.connect(user, password);
212            checkStoreForTestConnection(store);
213            final Folder f = store.getFolder(folderName);
214            f.open(Folder.READ_WRITE);
215    
216            final int msgCount = f.getMessageCount();
217    
218            final Message[] m = deleteCount == -1 ? f.getMessages() : f.getMessages(start, Math.min(msgCount, deleteCount + start - 1));
219            int d = 0;
220    
221            for (final Message message : m) {
222                message.setFlag(Flag.DELETED, true);
223                logger.info("Delete msgnum: {} with sid {}", message.getMessageNumber(), message.getSubject());
224                d++;
225            }
226    
227            f.close(true);
228            logger.info("Deleted " + d + " messages");
229            store.close();
230    
231        }

233        protected long getCount(final String index, final String type) {
234            logger.debug("getCount()");
235    
236            esSetup.client().admin().indices().refresh(new RefreshRequest()).actionGet();
237    
238            final CountResponse count = esSetup.client().count(new CountRequest(index).types(type)).actionGet();
239    
240            return count.getCount();
241        }

247        protected String loadFile(final String file) throws IOException {
248    
249            final StringWriter sw = new StringWriter();
250            IOUtils.copy(this.getClass().getResourceAsStream("/" + file), sw);
251            return sw.toString();
252    
253        }

255        protected void putMailInMailbox(final int messages) throws MessagingException {
256    
257            for (int i = 0; i < messages; i++) {
258                final MimeMessage message = new MimeMessage((Session) null);
259                message.setFrom(new InternetAddress(EMAIL_TO));
260                message.addRecipient(Message.RecipientType.TO, new InternetAddress(EMAIL_USER_ADDRESS));
261                message.setSubject(EMAIL_SUBJECT + "::" + i);
262                message.setText(EMAIL_TEXT + "::" + SID++);
263                message.setSentDate(new Date());
264                MockMailbox.get(EMAIL_USER_ADDRESS).getInbox().add(message);
265            }
266    
267            logger.info("Putted " + messages + " into mailbox");
268        }
269    
270        protected void registerRiver(final String typename, final String file) throws ElasticsearchException, IOException {
271            final IndexResponse res = esSetup.client().prepareIndex().setIndex("_river").setType(typename).setId("_meta")
272                    .setSource(loadFile(file)).execute().actionGet();
273            if (!res.isCreated()) {
274                throw new IOException("Unable to register river");
275            }
276        }

278        protected void renameMailbox(final Properties props, final String folderName, final String user, final String password)
279                throws MessagingException {
280            final Store store = Session.getInstance(props).getStore();
281    
282            store.connect(user, password);
283            checkStoreForTestConnection(store);
284            final Folder f = store.getFolder(folderName);
285    
286            f.renameTo(store.getFolder("renamed_from_" + folderName));
287    
288            logger.info("Renamed " + f.getFullName());
289            store.close();
290    
291        }

295            return new RiverSettings(ImmutableSettings.settingsBuilder().build(), XContentHelper.convertToMap(Streams.copyToByteArray(in),
296                    false).v2());
