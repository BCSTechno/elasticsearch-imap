033        @Test
034        public void testAttachments() throws Exception{
035
036                final RiverSettings settings = riverSettings("/river-imap-attachments.json");
037
038                final Properties props = new Properties();
039                final String user = XContentMapValues.nodeStringValue(settings.settings().get("user"), null);
040                final String password = XContentMapValues.nodeStringValue(settings.settings().get("password"), null);
041
042                for (final Map.Entry<String, Object> entry : settings.settings().entrySet()) {
043
044                        if (entry != null && entry.getKey().startsWith("mail.")) {
045                                props.setProperty(entry.getKey(), String.valueOf(entry.getValue()));
046                        }
047                }
048
049                registerRiver("imap_river", "river-imap-attachments.json");
050
051                final Session session = Session.getInstance(props);
052                final Store store = session.getStore();
053                store.connect(user, password);
054                checkStoreForTestConnection(store);
055                final Folder inbox = store.getFolder("INBOX");
056                inbox.open(Folder.READ_WRITE);
057
058
059
060                final MimeMessage message = new MimeMessage(session);
061                message.setFrom(new InternetAddress(EMAIL_TO));
062                message.addRecipient(Message.RecipientType.TO, new InternetAddress(EMAIL_USER_ADDRESS));
063                message.setSubject(EMAIL_SUBJECT + "::attachment test");
064                message.setSentDate(new Date());
065
066                BodyPart bp = new MimeBodyPart();
067                bp.setText("Text");
068                Multipart mp = new MimeMultipart();
069                mp.addBodyPart(bp);
070
071                bp = new MimeBodyPart();
072                DataSource ds = new ByteArrayDataSource(this.getClass().getResourceAsStream("/httpclient-tutorial.pdf"), "application/pdf");
073                bp.setDataHandler(new DataHandler(ds));
074                bp.setFileName("httpclient-tutorial.pdf");
075                mp.addBodyPart(bp);
076                message.setContent(mp);
077
078                inbox.appendMessages(new Message[]{message});
079                IMAPUtils.close(inbox);
080                IMAPUtils.close(store);
081
082                //let the river index
083                Thread.sleep(20*1000);
084
085                esSetup.client().admin().indices().refresh(new RefreshRequest()).actionGet();
086
087
088                SearchResponse get =  esSetup.client().prepareSearch("imapriverdata").setTypes("mail").execute().actionGet();
089                Assert.assertEquals(1, get.getHits().totalHits());
090
091                //BASE64 content httpclient-tutorial.pdf
092                Assert.assertTrue(get.getHits().hits()[0].getSourceAsString().contains("JVBERi0xLjQKJaqrrK0KNCAwIG9iag"));
093
094                get =  esSetup.client().prepareSearch("imapriverdata").addFields("attachments.content").setTypes("mail").setQuery(QueryBuilders.matchQuery("attachments.content", "wrapping")).execute().actionGet();//search(new SearchRequest("imapriverdata").types("mail")).actionGet();
095                Assert.assertEquals(1, get.getHits().totalHits());
096                Assert.assertEquals(1, get.getHits().hits()[0].field("attachments.content").getValues().size());
097                Assert.assertTrue(get.getHits().hits()[0].field("attachments.content").getValue().toString().contains("wrapping"));
098
099        }

181                Assert.assertEquals(1, get.getHits().totalHits());
182                Assert.assertEquals(2, get.getHits().hits()[0].field("attachments.content").getValues().size());
183
184                //second value is testWORD.docx
185                Assert.assertTrue(get.getHits().hits()[0].field("attachments.content").getValues().get(1).toString().contains("This paragraph is in the default text style"));
186
187        }
